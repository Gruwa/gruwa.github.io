• [title]: string - title text of dropdown (by default title is outputted in tooltip)
  • [size]: DropdownSizes = 'default' - different dropdown sizes ('default' | 'small' | 'big' | 'huge')
  • [type]: DropdownTypes = 'default' - different dropdown types ('default' | 'boxed' | 'primary')
  • [color]: DropdownColors = 'default' - dropdown head color ('default' | 'blue' | 'white' | 'grey')
  • [contentColor]: DropdownContentColors = 'default' - dropdown content background color ('default' | 'grey')
  • [iconName]: string = 'arrow-down' - custom dropdown icon
  • [placement]: DropdownPlacements = 'bottom-right' - where dropdown is appear ('top-left' | 'top-right' | 'bottom-left' | 'bottom-right')
  • [autoClose]: boolean = true - to close by clicking outside dropdown or by pressing ESC button
  • [isOpened]: boolean = false - to manually open dropdown
  • [closeOnChange]: boolean = false - should dropdown be closed, after ngModel change
  • [isCustomTooltipText]: boolean = false - is custom tooltip text?
  • [customTooltipText]: string - text for custom tooltip, could be HTML
  • (onToggle): EventEmitter<boolean> - trigger when dropdown is opened/closed
